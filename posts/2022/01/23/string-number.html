# 截取

var str = 'abcd';

// slice(start?, end?)：截取[start, end)区间即[start, end - 1]
// 参数均可选，默认start为0，end为length-1，即都不传则拷贝
// 均可取负，表示倒数，传入NaN会被转换为0
var str1 = str.slice();

// split(splitter, limit?)：原字符串分割成以splitter隔开的limit个子字符串组成数组并返回
// 参数均可选，默认splitter为undefined，limit为length-1
// splitter表示从原串中找到该分割符并以其进行分割，原串不存在匹配该正则的子串或字符串则相当于传入void 0
// 不传则默认不作分割且作为整个元素，类型为字符或正则
// imit表示从返回的数组中截取前limit个元素，从而限定返回的数组长度
// ['abcd']
var str21 = str.split(void 0);
// ['a', 'b', 'c']
var str22 = str.split('', 3);
// ['abcd']
var str23 = str.split(/\s*is\s*/);

// str.substr(start, length?)：从start开始截取length个字符即截取[start, start + length - 1]
// start可正可负，后者则表示倒数
// length默认为length
var str3 = str.substr(0);

// str.substring(start, end?)：截取[start, end - 1]索引之间的子串
// 相对于slice()至少要传起始索引，功能和后者一样，建议后者即可
var str4 = str.substring(0, 0);

# NaN

官方描述：NaN是一个放在global对象里的一个value。是一个代表Not-A-Number的value。

简单来说，NaN是一个value，这个vaule的type是number

# isNaN()

例子

isNaN(NaN); // true

isNaN('A String'); // true

isNaN(undefined); // true

isNaN({}); // true

这里不是NaN的value也被误判成了NaN，原因是isNaN会通过Number方法把参数转换成数字类型，如果转换成功就返回false，反之返回true。它只是判断参数是否能转成数字，不能用来判断是否严格等于NaN

# 浮点精度丢失

0.2+0.1!==0.3会发生三次精度丢失,示例代码

function judgeFloat(n, m) {
  // Number.prototype.toString()内置实现
  const binaryN = n.toString(2);
  const binaryM = m.toString(2);
  console.log(`${n}的二进制是    ${binaryN}`);
  console.log(`${m}的二进制是    ${binaryM}`);
  const MN = m + n;
  const accuracyMN = (m * 100 + n * 100) / 100;
  const binaryMN = MN.toString(2);
  const accuracyBinaryMN = accuracyMN.toString(2);
  console.log(`${n}+${m}的二进制是${binaryMN}`);
  console.log(`${accuracyMN}的二进制是    ${accuracyBinaryMN}`);
  console.log(`${n}+${m}的二进制再转成十进制是${to10(binaryMN)}`);
  console.log(`${accuracyMN}的二进制是再转成十进制是${to10(accuracyBinaryMN)}`);
  console.log(
    `${n}+${m}在js中计算是${to10(binaryMN) === to10(accuracyBinaryMN) ? '' : '不'}准确的`
  );
}
// 转十进制
function to10(n) {
  const pre = (n.split('.')[0] - 0).toString(2);
  const arr = n.split('.')[1].split('');
  let i = 0;
  let result = 0;
  while (i < arr.length) {
    result += arr[i] * Math.pow(2, -(i + 1));
    i++;
  }
  return result;
}
judgeFloat(0.1, 0.2);
judgeFloat(0.6, 0.7);

计算机中所有的数据都是以二进制存储的,所以在计算时计算机要把数据先转换成二进制进行计算,然后再把计算的结果转换成十进制

所以在计算0.1+0.2时,二进制计算发生了精度丢失,导致再转换成十进制后和预期的结果不符

# 二进制小数的存储方式

小数的二进制大多数都是无限循环的,ECMAScript中的Number类型遵循IEEE 754标准,使用64位固定长度来表示

IEEE754标准包含一组实数的二进制表示法,它有三部分组成:

  • 符号位

  • 指数位

  • 尾数位

JS采用的是64位双精度浮点数编码,所以它的符号位占1位,指数位占11位,尾数位占52位.

解释一下啥是符号位 指数位 尾数位

以0.1为例子,它的二进制是0.0001100110011001100...

为了节省空间,在计算机中它是以科学技术法表示的,也就是

1.100110011001100... X 2^-4

以十进制的数理解一下:1100的科学计数法为11 X 10^2

  • 符号位:表示正负,1为正,0为负

  • 指数位:存储科学计数法的指数

  • 尾数位:存储科学计数法后的有效数字

所以我们通常看到的二进制,其实是计算机实际存储的尾数位

# js中的toString(2)

由于尾数位只能存储52个数字,这就能解释toString(2)的结果了

如果计算机没有存储空间的限制,那么0.1的二进制应该是

0.00011001100110011001100110011001100110011001100110011001...

科学计数法尾数位

1.1001100110011001100110011001100110011001100110011001...

但是由于限制,有效数字第53位以及后面的数字不能存储,如果是1则进1,0则舍弃

0.1的二进制科学计数法的第53位是1,所以进1

接着再恢复回去4位0,就能得到最终js的二进制结果了

0.0001100110011001100110011001100110011001100110011001101

0.2也有着同样的问题,正是由于这样的存储导致精度的丢失

# 解决方法

  1. 将数字转成整数
  2. ES6的bigInt

# 大数

# JS最大数

不考虑BigInt

由于IEEE754双精度64位规范的限制:

指数位能表示的最大数字:1023

尾数位能表达的最大数字即尾数位都为1的情况

所以JS能表示的最大数字即

1.111...X 2^1023 这个结果转换成十进制是1.7976931348623157e+308,这个结果即为Number.MAX_VALUE

# 最大安全数

JS中的Number.MAX_SAFE_INTEGER表示最大安全数字,计算结果是9007199254740991,在这个范围内不会出现精度丢失(小数除外),这个数实际上就是1.111...X 2^52。

# 相加

对应leetcode <字符串相加>

方法较为简单,本质上两种方法基本差不多

方法一是将大数字符串转为数组

var addStrings = function(num1, num2) {
    let stack1=num1.split("")
    let stack2=num2.split("")
    let res=[]
    let carry=0
    while(stack1.length>0||stack2.length>0||carry>0){
        let val=carry
        val+=stack1.length>0?Number(stack1.pop()):0
        val+=stack2.length>0?Number(stack2.pop()):0
        carry=parseInt(val/10)
        val=val%10

        res.unshift(val)
    }
    return res.join("")
};

方法二是补全较小数前面的0

var addStrings = function(num1, num2) {
    let len=Math.max(num1.length,num2.length)
    num1=num1.padStart(len,"0")
    num2=num2.padStart(len,"0")
    let carry=0
    let res=[]
    for(let i=len-1;i>=0;i--){
        let val=carry
        val+=(Number(num1[i])+Number(num2[i]))
        carry=parseInt(val/10)
        val=val%10
        res.unshift(val)
    }
    carry>0?res.unshift(carry):res
    return res.join("")
};

# 相乘

对应leetcode 《字符串相乘》

需要琢磨一下

参考链接：https://mp.weixin.qq.com/s/GFg9wBmbOVGP_d2TVkBZIg

var multiply = (num1, num2) => {
  // 1. 参数校验，同理大数相加
  if (isNaN(num1) || isNaN(num2)) return '';
  if (num1 === '0' || num2 === '0') return '0';

  const l1 = num1.length;
  const l2 = num2.length;
  const res = [];

  // 2. 两次循环
  for (let i = l1 - 1; i >= 0; i--) {
    for (let j = l2 - 1; j >= 0; j--) {
      // 3. 假设两个四位数相乘其结果最多为八位即其high为索引最大为8，high和low位的索引分别为idx2和1
      const idx1 = i + j;
      const idx2 = idx1 + 1;
      // 4. 当前位数相乘结果需要加上上一位结果的high位，相乘运算符隐式转为Number
      const product = num1[i] * num2[j] + (res[idx2] || 0);
      // 5. 高位的计算需要加上可能需要补齐的0
      const high = ~~(product / 10) + (res[idx1] || 0);
      const low = product % 10;
      // 6. 分别存区分的两位入结果集
      res[idx2] = low;
      res[idx1] = high;
    }
  }
  // 7. 剔除结果可能存在的0因为已在步骤四为结果集的开头补齐0
  return res.join('').replace(/^0+/, '');
};

// 1 845 756
console.log(multiply('198', '9322'));
